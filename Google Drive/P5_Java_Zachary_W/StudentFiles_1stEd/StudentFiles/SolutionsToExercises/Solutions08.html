6. (a) public static boolean isPrime(int n) { if (n < 3) return n == 2; else if (n % 2 == 0) return false; int m = 3; while (m * m <= n) { if (n % m == 0) return false; m += 2; } return true; }
(b) public static boolean isPrime(int n) { if (n < 5) return n == 2 || n == 3; else if (n % 2 == 0 || n % 3 == 0) return false; int m = 5; while (m * m <= n) { if (n % m == 0 || n % (m + 2) == 0) return false; m += 6; } return true; }
7.   public static boolean isPerfectSquare(int n) { int p = 1, sum = 0; while (sum < n) { sum += p; p += 2; } return sum == n; }
